当token不存在时 跳转到登陆界面

只有登录界面时 (不考虑注册等其他界面)

  let token=localStorage.getItem("token");
  if(token){//判断是否登录
    next()
  }

  else{
    if(to.path!=='/login'){
      alert("请先登录")
      next({path:'/login'})
    }
    else{
      next()
    }
  }
});
